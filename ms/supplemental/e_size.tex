\section{E size}

 \begin{equation}
 E_{size} = \frac{\sum_{i = 1}^n c_i^2}{ \sum_{i = 1}^n c_i } = \frac{\frac{\sum_{i = 1}^n c_i^2}{n}}{ \frac{\sum_{i = 1}^n c_i}{n} } =? \\
 \frac{\hat{E[C^2]}}{ \hat{E[C]} } 
 \end{equation}
